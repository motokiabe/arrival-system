async function checkSignal() { const username = 'motokiabe'; // raw.githubusercontent ではなく、API経由で直接「最新」を取得するように変更 const url = `https://api.github.com/repos/${username}/arrival-system/contents/signal.json?t=` + new Date().getTime(); try { const res = await fetch(url); const fileData = await res.json(); // GitHub APIは中身をBase64で返すので、それを解読します const content = JSON.parse(atob(fileData.content)); if (content.status === 'start') { startTimer(); } } catch (e) { console.log("チェック失敗: ", e); } }
